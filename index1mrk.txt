    // 配列操作関数
    function shuffleArray(array) {
        for (let i = array.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [array[i], array[j]] = [array[j], array[i]];
        }
        return array;
    }

    // 問題文のパース（テキスト形式への対応）
    function parseQuestion(qData) {
        // もしすでに数値のaなら、indexに直す(1始まり->0始まり)
        let ansIdx = (typeof qData.a === 'number') ? qData.a - 1 : parseInt(qData.a) - 1;

        // 改行で選択肢が分かれている形式の場合の処理
        // (JSON形式で記述されている場合はそのまま利用)
        let options = ["1", "2", "3", "4", "5"]; // デフォルト
        let text = qData.q;

        // 問題文中に選択肢が含まれている場合の簡易パース（これまでの形式対応）
        if(qData.q.includes("1.")){
             const parts = qData.q.split(/\n\n1\./); 
             if(parts.length > 1){
                 text = parts[0];
                 const optionsText = "1." + parts[1];
                 const tempOpts = optionsText.split(/\n/).map(o => o.replace(/^\d+\.\s*/, '').trim()).filter(o => o);
                 if(tempOpts.length >= 2) options = tempOpts;
             }
        }
        
        return { text: text, options: options, answerIdx: ansIdx, exp: qData.exp };
    }
